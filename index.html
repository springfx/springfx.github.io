The Spring Framework meets JavaFX

A simple application

class NonameApplication extends Application {

    @Autowired
    FXMLLoader fxmlLoader

    ApplicationContext applicationContext

    @Override
    void start(Stage primaryStage) {
        // Create an application context of your choice...
        applicationContext = new ClassPathXmlApplicationContext(['application-context.xml'] as String[], false)

        // ...and bind it to the current JavaFX thread
        ApplicationContextHolder.bindContext(applicationContext, this, primaryStage)

        // Ready to use auto wiring and other Spring features
        fxmlLoader.load(...)

        primaryStage.show()
    }

    @Override
    void stop() throws Exception {
        ApplicationContextUtils.closeContext(applicationContext)
    }

    static void main(String[] args) {
        launch(NonameApplication, args)
    }
}
